def fix_overlapping_events(events):
    """Add 1 to current event's sample time if it overlaps with the previous one"""
    old_t = -1000
    for ii, line in enumerate(events):
        new_t = line[0]
        if old_t == new_t:
            events[ii, 0] += 1
        old_t = new_t
    return events

def make_raw(name, exp="vis_scene", save=True, lp=.1, hp=40, sfreq=sfreq):
    """Read in, filter, resample unprocessed data, drop M1/M2 if possible, store as fif"""
    fname = in_template.format(name, exp)
    raw = mne.io.read_raw_brainvision(
        fname, event_id={"actiCAP Data On": 9999},  # just to suppress warnings
        montage=montage, preload=True, eog=['SO1', 'SO2'])
    events = fix_overlapping_events(mne.find_events(raw, shortest_event=0, consecutive=True))
    
    # crop 10 seconds before the first event
    tmin = raw.times[events[0][0]]-10
    if tmin < 0: # if tmin is negative just set it at the beginning
        tmin = raw.times[0] 
    # crop 10 seconds after the last event
    tmax = raw.times[events[-1][0]]+10
    if tmax > raw.times[-1]: # if tmax is bigger then the max time, just take that
        tmax = raw.times[-1]
    
    # crop it
    raw.crop(tmin=tmin, tmax=tmax)
    
    try:
        raw.filter(lp, hp, n_jobs='cuda', fir_design="firwin")
        raw, events = raw.resample(sfreq, npad='auto', n_jobs='cuda', events=events)
    except Exception:
        print("CUDA failed for " + name)
        raw.filter(lp, hp, n_jobs=-1, fir_design="firwin")
        raw, events = raw.resample(sfreq, npad='auto', n_jobs=-1, events=events)
    try:
        raw.drop_channels(["M1", "M2"])
    except:
        pass
    if save:
        raw.save(out_template.format(name, exp, "raw"), overwrite=True)
        mne.write_events(out_template.format(name, exp, "eve"), events)

def make_ica(name, n_components=60, filter_=(8, None)):
    """Joint ICA for both experimens"""
    exp1, exp2 = exps
    raw1 = mne.io.read_raw_fif(out_template.format(name, exp1, "raw"), preload=True)
    raw2 = mne.io.read_raw_fif(out_template.format(name, exp2, "raw"), preload=True)
    raw = mne.concatenate_raws(
        [raw1.filter(*filter_, n_jobs="cuda"), raw2.filter(*filter_, n_jobs="cuda")])
    del raw1
    del raw2
    ica = mne.preprocessing.ICA(n_components=n_components).fit(raw)
    ica.exclude, _ = ica.find_bads_eog(raw)
    ica.plot_components(ica.exclude, outlines = "skirt")
    ica.save(out_template.format(name, "all", "ica"))
    return ica

def get_epochs_for_ar(name, exp="vis_scene", version="all", filter_=False, tmin=-.2, tmax=.9):
    """Get epochs (scene data)"""
    raw = mne.io.read_raw_fif(out_template.format(name, exp, "raw"), preload=True)
    if filter_ is not False:
        try:
            raw.filter(*filter_, n_jobs='cuda')
        except Exception:
            raw.filter(*filter_, n_jobs=-2)
    ica = mne.preprocessing.read_ica(out_template.format(name, version, "ica"))
    raw = ica.apply(raw, exclude=ica.exclude).pick_types(eeg=True)
    events = mne.read_events(out_template.format(name, exp, "eve"))
    epochs = mne.Epochs(raw, events, event_ids["vis_scene"], tmin=tmin, tmax=tmax,
                        reject=None, preload=True)
    epochs.apply_baseline((None, 0))
    return epochs

df = pd.read_csv(sourceloc + "csv/lookUp320_2.csv", index_col=0)

df["time"] = (df["onset"] * sfreq).astype(int)
df["item"] = df.index.map(lambda x: int(x.split("_")[-1]))
df = df.set_index(["trig", "item"])[["time"]]

def make_events_sentences(events, df=df):
    """Extract timings for the audio N400"""
    df_events = pd.DataFrame()
    inds = np.where(np.in1d(events[:, -1], [200, 201, 210, 211]))[0]  # find crit conds
    df_events["item"] = events[inds-1, -1] -1  # find preceding trg for item code
    df_events["trig"] = events[inds, -1]  # find the actual trg (condition code)
    df_events["time"] = events[inds, 0] + delay  # find the time, add delay (in samples)
    df_events = df_events.set_index(["trig", "item"])

    # merge with orig timeframe on index = (trg, item), and add the times (both in samples)
    df_events_fin = (df_events["time"] + df["time"]).dropna().reset_index(
    ).sort_values("time")[["time", "item", "trig"]]  # sort to avoid a warning
    return df_events_fin.values.astype(int)


def get_epochs(name, exp="sentence", ar=True, baseline=(None, 0),
               filter_=(None, 20), tmin=-.2, tmax=.9):
    """Extract epochs"""
    raw = mne.io.read_raw_fif(out_template.format(name, exp, "raw"), preload=True)

    # should we filter?
    if filter_ is not False:
        try:
            raw.filter(*filter_, n_jobs='cuda')
        except Exception:
            raw.filter(*filter_, n_jobs=-2)

    # apply ica
    ica = mne.preprocessing.read_ica(out_template.format(name, "all_corrmap", "ica"))
    raw = ica.apply(raw, exclude=ica.exclude).pick_types(eeg=True)

    # do events and epoch
    events = mne.read_events(out_template.format(name, exp, "eve"))
    if exp == "sentence":
        events = make_events_sentences(events)
    epochs = mne.Epochs(raw, events, event_ids[exp], tmin=tmin, tmax=tmax, reject=None)

    # apply autorej
    if ar is True:
        with open(out_template.format(name, "vis_scene", "ar"), "rb") as f:
            epochs = pickle.load(f).transform(epochs.load_data().apply_baseline((None, 0)))
    
    # apply baseline
    if baseline is not None:
        epochs.apply_baseline(baseline)

    return epochs
